#include <Wire.h>
#include <MPU6050.h>

MPU6050 mpu;

// Define ESP32 I2C pins
#define I2C_SDA 27
#define I2C_SCL 14

void setup() {
  Serial.begin(115200);

  // Start I2C with defined pins
  Wire.begin(I2C_SDA, I2C_SCL);

  Serial.println("Initializing MPU6050...");
  mpu.initialize();

  if (mpu.testConnection()) {
    Serial.println("MPU6050 connection successful!");
  } else {
    Serial.println("MPU6050 connection failed!");
    while (1);
  }
}

void loop() {
  int16_t gx, gy, gz;

  // Read raw gyro values
  mpu.getRotation(&gx, &gy, &gz);

  // Convert raw values to deg/sec (default sensitivity: 131 LSB/(°/s) for ±250°/s)
  float gyroX = gx / 131.0;
  float gyroY = gy / 131.0;
  float gyroZ = gz / 131.0;

  Serial.print("Gyro X: "); Serial.print(gyroX); Serial.print(" °/s, ");
  Serial.print("Gyro Y: "); Serial.print(gyroY); Serial.print(" °/s, ");
  Serial.print("Gyro Z: "); Serial.print(gyroZ); Serial.println(" °/s");

  delay(200);
}
